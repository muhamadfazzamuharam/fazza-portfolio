// Regression for BUG-01 — discount must apply to subtotal, not shipping
pm.test('Voucher discount applies to subtotal, not shipping', () => {
  pm.expect(order.discount.appliedTo).to.eql('subtotal');
  pm.expect(order.shippingFee).to.eql(order.shippingFeeOriginal);
});

pm.test('Grand total = subtotal - discount + shipping', () => {
  const expected = order.subtotal - order.discount.amount + order.shippingFee;
  pm.expect(order.grandTotal).to.eql(expected);
});

pm.test('Order status is AWAITING_PAYMENT', () => {
  pm.expect(order.status).to.eql('AWAITING_PAYMENT');
});

Sample code

lapakku-login.spec.js — page object + happy-path test 
class LoginPage {
  constructor(driver) {
    this.driver = driver;
    this.locators = {
      emailInput: By.css('[data-testid="login-email"]'),
      passwordInput: By.css('[data-testid="login-password"]'),
      submitBtn: By.css('[data-testid="login-submit"]'),
    };
  }

  async login(email, password) {
    await this.driver.findElement(this.locators.emailInput).sendKeys(email);
    await this.driver.findElement(this.locators.passwordInput).sendKeys(password);
    await this.driver.findElement(this.locators.submitBtn).click();
  }
}

it('TC-01 redirects to /home when credentials are valid', async () => {
  await loginPage.login('qa-tester@lapakku.dev', 'Qa!Test1234');
  await driver.wait(until.urlContains('/home'), 10000);
  expect(await driver.getCurrentUrl()).to.include('/home');
});
